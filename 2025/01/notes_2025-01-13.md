SELECT
    a.id AS a_id,
    b.id,
    b.bioreg_id,
    a.antibody_name,
    b.name as b_name

    (
        SELECT
            b.date_created,
            b.id,
            b.bioreg_id,
            a.child_id,
            b.name AS antibody_name

            c2c.complex_child a
        JOIN c2c.complex b ON b.id = a.complex_id

    SELECT *
    FROM
        c2c.complex_child a
    JOIN c2c.complex b ON b.id = a.child_id
    WHERE
        upper(composition) NOT LIKE '%DELETED%'
) b ON b.child_id = a.child_id
