```sql
-- use XMLAGG
SELECT RTRIM(
  XMLAGG(
    XMLELEMENT(e, experiment_id || ', ') 
    ORDER BY experiment_id
  ).GETCLOBVAL(), 
  ', '
) AS experiment_ids
FROM (
SELECT experiment_id
FROM eln_writeup
WHERE 
  DBMS_LOB.INSTR(write_up, '?') > 0
  AND REGEXP_LIKE(
    write_up,
    '(\(|^|\s)\?\s*([A-Za-z0-9.-]+|\(|\))', 
    'i'
  )
)
;

-- use listagg
SELECT LISTAGG(experiment_id, ', ') WITHIN GROUP (ORDER BY experiment_id) AS experiment_ids
FROM (
SELECT experiment_id
FROM eln_writeup
WHERE 
  DBMS_LOB.INSTR(write_up, '?') > 0
  AND REGEXP_LIKE(
    write_up,
    '(\(|^|\s)\?\s*([A-Za-z0-9.-]+|\(|\))', 
    'i'
  )
)
```
