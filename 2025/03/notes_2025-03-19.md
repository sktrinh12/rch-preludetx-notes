SELECT 
    "SQH_DS",
    "SQH_TS"
FROM (
    SELECT 
        "SQH_ID", 
        "SQH_DS", 
        "SQH_USER", 
        "SQH_TS", 
        "SQH_STATUS", 
        "SQH_TEXT",
        ROW_NUMBER() OVER (
            PARTITION BY "SQH_DS" 
            ORDER BY "SQH_TS" DESC
        ) AS rn
    FROM DS3_APPDATA.BRO_DS_SQL_HISTORY
) ranked
WHERE rn = 1
AND DBMS_LOB.INSTR(UPPER(SQH_TEXT), 'IC50_NEW_RESULTS_SUMMARY') > 0 
ORDER BY "SQH_DS"
;
