    formatted_id,
    CASE
        WHEN compound_status IS NULL THEN
            POWER(10, AVG(CASE WHEN ic50 > 0 THEN LOG(10, ic50 * 1000) ELSE NULL END))

            MAX(ic50 * 1000)

    CASE
        WHEN compound_status IS NULL THEN
