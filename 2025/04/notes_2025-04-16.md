#### python script to generate sql text
- idea behind this was to generate all the CTE expressions for each PRT# and their corresponding column names. this was semi-automated, however far too manual still, lots of typing and copying/pasting
    * run `python3 sql_jak2_check.py --mode listagg --csv input.csv` first from the list of `formatted_id,cell_line` csv file to generate each sql query that generates the column lists

    * `python3 sql_jak2_check.py --mode compare --columns "CS_IC50_SET2, DMAX_SET2, HC_SET2, IC50_NM_CTG_SET2, IC50_NM_SET2, N_IC50_CTG_SET2, N_IC50_SET2, RATIO_ABS_IC50_F36P_SET2, RATIO_ABS_IC50_TF1_SET2, RATIO_ABS_IC50_UT7_SET2, RESP_HC_CTG_SET2, RESP_HC_SET2, SD_IC50_CTG_SET2, SD_IC50_SET2, SET2_ABS_IC50, SET2_CTG_ABS_IC50, SPAN_SET2" --id PRT1011776`

##### first script for prototype testing semi-automatic, however too slow
```py
import csv
import argparse
from collections import defaultdict

def normalize_cell_line(name):
    name = name.upper()
    if '-' in name:
        parts = name.split('-')
        if len(name) <= 6:
            return name.replace('-', '')
        else:
            return parts[0]
    return name

def extract_cell_lines_by_id(csv_file):
    cell_lines_by_id = defaultdict(set)
    with open(csv_file, newline='') as f:
        reader = csv.DictReader(f)
        for row in reader:
            formatted_id = row['formatted_id']
            cell_line = normalize_cell_line(row['cell_line'])
            cell_lines_by_id[formatted_id].add(cell_line)
    return cell_lines_by_id

def generate_listagg_sql_queries(cell_lines_by_id):
    queries = []
    for formatted_id, cell_lines in sorted(cell_lines_by_id.items()):
        queries.append(f"-- formatted_id: {formatted_id}")
        for cell_line in sorted(cell_lines):
            query = f"""SELECT LISTAGG(column_name, ', ') WITHIN GROUP (ORDER BY column_name) AS columns
FROM all_tab_columns
WHERE table_name = 'SUMM_JAK2'
  AND REGEXP_LIKE(column_name, '{cell_line}', 'i')"""
            queries.append(query)
    return queries

def generate_comparison_query(columns_csv: str, formatted_id: str):
    columns = [col.strip() for col in columns_csv.split(',')]
    select_lines = ["SELECT\n  a.formatted_id,"]
    
    for col in columns:
        alias = col.lower() + "_tmp"
        select_lines.append(f"  a.{col} AS {alias},")
        alias = col.lower() + "_main"
        select_lines.append(f"  b.{col} AS {alias},")
    
    # Remove last comma
    select_lines[-1] = select_lines[-1].rstrip(',')

    join_block = f"""
FROM tmp_summ_jak2 a
JOIN summ_jak2 b ON a.formatted_id = b.formatted_id
WHERE a.formatted_id = '{formatted_id}'"""

    return '\n'.join(select_lines) + join_block


if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Generate SQL queries from CSV or column list")
    parser.add_argument("--mode", choices=["listagg", "compare"], required=True, help="Choose the mode to run")
    parser.add_argument("--csv", help="Path to CSV file for listagg mode")
    parser.add_argument("--columns", help="Comma-separated column names for compare mode")
    parser.add_argument("--id", help="Formatted ID for compare mode")

    args = parser.parse_args()

    if args.mode == "listagg":
        if not args.csv:
            raise ValueError("You must provide --csv when using mode 'listagg'")
        cell_lines_by_id = extract_cell_lines_by_id(args.csv)
        print("=== SQL Listagg Queries ===\n")
        for q in generate_listagg_sql_queries(cell_lines_by_id):
            print(q)
            print()

    elif args.mode == "compare":
        if not args.columns or not args.id:
            raise ValueError("You must provide both --columns and --id for compare mode")
        comparison_query = generate_comparison_query(args.columns, args.id)
        print("=== SQL Comparison Query ===\n")
        print(comparison_query)
```

##### Second python script to generate CTE query quicker and only two cell lines
- the result of this python file generated **~12,000** lines of SQL code to be used as the DS, `TMP_OUTPUT_JAK2_PRTN` in *tall* table format for comparison of two identical columns
```py
formatted_ids = [
    'PRT1000816',
    #... same from csv file below 
]

sets = {
    'SET2': [
        'CS_IC50_SET2', 'DMAX_SET2', 'HC_SET2', 'IC50_NM_CTG_SET2', 'IC50_NM_SET2',
        'N_IC50_CTG_SET2', 'N_IC50_SET2', 'RATIO_ABS_IC50_F36P_SET2', 'RATIO_ABS_IC50_TF1_SET2',
        'RATIO_ABS_IC50_UT7_SET2', 'RESP_HC_CTG_SET2', 'RESP_HC_SET2',
        'SD_IC50_CTG_SET2', 'SD_IC50_SET2', 'SET2_ABS_IC50', 'SET2_CTG_ABS_IC50', 'SPAN_SET2'
    ],
    'TF1': [
        'CS_IC50_TF1', 'DMAX_TF1', 'HC_TF1', 'IC50_NM_CTG_TF1', 'IC50_NM_TF1',
        'N_IC50_CTG_TF1', 'N_IC50_TF1', 'RATIO_ABS_IC50_TF1_SET2',
        'RESP_HC_CTG_TF1', 'RESP_HC_TF1', 'SD_IC50_CTG_TF1', 'SD_IC50_TF1',
        'SPAN_TF1', 'TF1_ABS_IC50', 'TF1_CTG_ABS_IC50'
    ]
}

i_break = 10 # this is for testing; set to a higher integer than number of prts from csv file
output_ctes = []
output_selects = []

for i, formatted_id in enumerate(formatted_ids, start=1):
    if i == i_break:
        break
    set_type = 'SET2' if i % 2 == 0 else 'TF1'
    cte_name = f"t{i}"
    fields = sets[set_type]
    
    cte = [f"{cte_name} AS (",
           "  SELECT",
           "    a.formatted_id,"]

    for field in fields:
        tmp_col = f"    a.{field} AS {field.lower()}_tmp,"
        main_col = f"    b.{field} AS {field.lower()}_main,"
        cte.extend([tmp_col, main_col])

    # Remove trailing comma from last two lines
    cte[-1] = cte[-1].rstrip(',')
    cte.append("  FROM tmp_summ_jak2 a")
    cte.append("  JOIN summ_jak2 b ON a.formatted_id = b.formatted_id")
    cte.append(f"  WHERE a.formatted_id = '{formatted_id}'")
    cte.append(")")

    if i == 1:
        cte[0] = "WITH " + cte[0]   

    output_ctes.append("\n".join(cte))
    
    # Build SELECT block
    for field in fields:
        short = field.lower()
        if i == 1 and field == fields[0]:
            select_tmp = f"SELECT formatted_id AS FORMATTED_ID, '{short}' AS MEASUREMENT, 'tmp' AS SOURCE, CAST({short}_tmp AS VARCHAR2(100)) AS VALUE FROM {cte_name}"
            select_main = f"SELECT formatted_id AS FORMATTED_ID, '{short}' AS MEASUREMENT, 'main' AS SOURCE, CAST({short}_main AS VARCHAR2(100)) AS VALUE FROM {cte_name}"
        else:
            select_tmp = f"SELECT formatted_id, '{short}', 'tmp', CAST({short}_tmp AS VARCHAR2(100)) FROM {cte_name}"
            select_main = f"SELECT formatted_id, '{short}', 'main', CAST({short}_main AS VARCHAR2(100)) FROM {cte_name}"
        
        output_selects.append(select_tmp)
        output_selects.append(select_main)

print(",\n\n".join(output_ctes))
print("\n\n" + "\nUNION ALL\n".join(output_selects))
```
